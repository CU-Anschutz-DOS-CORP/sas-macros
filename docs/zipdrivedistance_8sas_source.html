1/**
2 @file
3 @brief Gets driving distance between zip codes
4
5 @details
6 @par Purpose
7 This macro accesses Google Maps and returns the driving distance between
8 two 5-digit zip codes
9
10 @author Rocio Lopez
11
12 @date 10-15-2018
13
14 @version SAS 9.4
15
16 @par Usage
17 A call to this macro has some or all of these parameters (defaults given):
18 @code{.sas}
19 %zipDriveDistance(
20 data=
21 ,zipcode1=
22 ,zipcode2=
23 ,outdata =
24 ,straightline=1
25 ,debug=0
26 );
27 @endcode
28
29 @returns
30 A sas data set with columns added for driving distance, driving distance units,
31 driving time, and, if requested, straight line distance
32
33 @note
34 @parblock
35 @li This is based on
36 <a href="http://support.sas.com/resources/papers/proceedings10/050-2010.pdf"
37 target="_blank"> SGF Paper 050-2010</a> and
38 <a href="https://communities.sas.com/t5/SAS-Communities-Library/Driving-Distances-and-Drive-Times-using-SAS-and-Google-Maps/ta-p/475839"
39 target="_blank"> this SAS communities post</a>
40
41 @li Zip codes should be 5 digits but can be numeric or character
42
43 @li Finding the desired values of distance within the web page produced by Google Maps
44 is based on the current structure of those web pages; these structure is subject to
45 change at any time and this macro will probably need to be updated to accommodate
46 any changes
47
48 @li Takes 1-3 seconds per query (observation)
49
50 @li The user should revise observations without reported distances as sometimes
51 they are returned as blank but a manual check of Google will give a distance
52 @endparblock
53
54 @param data Name of SAS dataset containing zip codes
55
56 @param zipCode1 Name of column with zip code of origin
57
58 @param zipCode2 Name of column with destination zip code
59
60 @param straightLine Include the straight line distance (ZIPCITYDISTANCE function)?
61 @n Allowed options are:
62 @n 0 for no
63 @n 1 for yes (DEFAULT)
64
65 @param outData Name of new SAS dataset that will have original data + driving distance
66 @n If not provided, default will be <data>_drive_distance
67
68 @param debug Debugging on/off
69 @n Allowed options are:
70 @n 0 for off (DEFAULT)
71 @n 1 for on
72
73 @par Example(s)
74 @code{.sas}
75 data zip_info;
76 input z1 @@;
77 if z1=00966 then z2=00969;
78 else z2=12203;
79 format z1 z2 z5.;
80 datalines;
81 02138 13502 20037 94117 12144 12203 00966
82 ;
83 run;
84
85 %zipDriveDistance(
86 data=zip_info
87 ,zipcode1=z1
88 ,zipcode2=z2
89 );
90
91 proc print data=zip_info_drive_distance; run;
92 @endcode
93
94 @par Revision History
95 @b 12-01-2023 Updated URL call to exclude comma between city and state
96 @n @b 03-07-2023 Skipping zip codes not found in sashelp.zipcode (need zip+city for search)
97 @n @b 03-06-2023 Optimized the way the data is read in and processed
98 @n @b 03-02-2023 Output dataset now includes drive time in seconds (with time6. format)
99 @n @b 05-15-2019 Updated to also search for distances given in kilometers,
100 these are transformed into miles
101**/

103%macro zipDriveDistance(
104 data= ,
105 zipcode1= ,
106 zipcode2= ,
107 outdata = ,
108 straightline=1,
109 debug=0);
110
111%*****************************************************************************;
112%* ERROR CHECKING ;
113%*****************************************************************************;
114%let errorflag=0;
115options minoperator;
116
117%* Check debug option and turn on macro checking options;
118%if not(%upcase(&debug) in (0 1)) %then %do;
119 %put WARNING: &debug must be either 0 or 1.;
120 %put WARNING: DEBUG = **&debug**;
121 %put WARNING: Default value of 0 will be used for DEGUG;
122 %let degug=0;
123%end; %*end to debug check;
124
125%if &debug eq 1 %then %do;
126 options mlogic mlogicnest varinitchk=warn mergenoby=warn;
127%end; %* end to debug eq 1;
128
129%* Check DATA is given and exists;
130%if %length(&data) = 0 %then %do;
131 %PUT ERROR: Required parameter DATA not provided. Macro will stop executing;
132 %LET ERRORFLAG=1;
133%end; %*end to if length(data) = 0;
134
135%if %length(&data) > 0 and %SYSFUNC(exist(&data)) ne 1 %then %do;
136 %PUT ERROR: Data set %upcase(&data) does not exist. Macro will stop executing;
137 %LET ERRORFLAG=1;
138%end; %*end to if length(data) ne 0;
139
140%* Check that both zipcodes are proived and exist on the database;
141%if %length(&zipcode1) = 0 %then %do;
142 %put ERROR: Required parameter ZIPCODE1 is not provided. Macro will stop executing.;
143 %let ERRORFLAG=1;
144%end; **end to if length(zipcode1) = 0 ;
145
146%if %length(&zipcode2) = 0 %then %do;
147 %put ERROR: Required parameter ZIPCODE2 is not provided. Macro will stop executing.;
148 %let ERRORFLAG=1;
149%end; **end to if length(zipcode2) = 0 ;
150
151%if %length(&data) > 0 and %SYSFUNC(exist(&data))=1 and %length(&zipcode1) > 0 %then %do;
152
153 %* Check if VAR exists;
154 %let dsid=%sysfunc(open(&data));
155 %let exist1=%sysfunc(varnum(&DSID, &zipcode1));
156 %let vartype1=%sysfunc(vartype(&DSID, &exist1));
157 %let exist2=%sysfunc(varnum(&DSID, &zipcode2));
158 %let vartype2=%sysfunc(vartype(&DSID, &exist2));
159 %let rc=%sysfunc(close(&DSID));
160
161 %* both must exist;
162 %if &exist1 eq 0 %then %do;
163 %put ERROR: Variable %upcase(&zipcode1) does not exist in dataset %upcase(&data). Macro will stop executing.;
164 %let ERRORFLAG=1;
165 %end; **end to if Exist1=0 then do;
166
167 %if &exist2 eq 0 %then %do;
168 %put ERROR: Variable %upcase(&zipcode2) does not exist in dataset %upcase(&data). Macro will stop executing.;
169 %let ERRORFLAG=1;
170 %end; **end to if Exist1=0 then do;
171
172%end; ** end to check zipcodes exit;
173
174%* If outdata not provided create default;
175%if %length(&outdata) eq 0 %then %let outdata=&data._drive_distance;
176
177%* Check straightline;
178%if not(%upcase(&straightline) in (0 1 YES NO Y N)) %then %do;
179 %put WARNING: STRAIGHTLINE must be either 0, 1, no or yes.;
180 %put STRAIGHTLINE = **&straightline**;
181 %put Defuault value of STRAIGHTLINE = 1 (yes) will be used instead.;
182%end; **end to if straightline not in ();
183
184%*****************************************************************************;
185%* IF ANY ERRORS FOUND THEN ABORT EXECUTION ;
186%*****************************************************************************;
187%if &errorflag eq 1 %then %do;
188 data _null_;
189 abort 3;
190 run;
191%end;

193%*****************************************************************************;
194%* DELETE ANY DATA SET NAMED _DISTANCE_ THAT MIGHT EXIST IN THE WORK LIBRARY ;
195%*****************************************************************************;
196proc datasets lib=work nolist;
197 delete _distance_;
198quit;
199
200%*****************************************************************************;
201%* OTHERWISE PROCEED TO CALCULATION ;
202%*****************************************************************************;
203%* Get unique combinations of zips;
204proc freq data=&data noprint;
205 where
206 %if &vartype1 eq N %then %do; put(&zipcode1, z5.) %end;
207 %else %if &vartype1 eq C %then %do; put(&zipcode1, $5.) %end;
208 ne
209 %if &vartype2 eq N %then %do; put(&zipcode2, z5.) %end;
210 %else %if &vartype2 eq C %then %do; put(&zipcode2, $5.) %end; ;
211 tables &zipcode1*&zipcode2/list out=_zips_;
212run;
213
214%* Place number of zip in a macro variable;
215data _null_;
216 call symputx('nzips',obs);
217 stop;
218 set _zips_ nobs=obs;
219run;
220
221%* Create a loop to access Google Maps multiple time;
222%let zipcity1 = ;
223%let zipcity2 = ;
224%do j=1 %to &nzips;
225 data _null_;
226 nrec = &j;
227 set _zips_ point=nrec;
228
229 %* Make sure they are both the same type;
230 %if &vartype1 eq N %then %do;
231 __zip1=put(&zipcode1,z5.);
232 %end;
233 %else %if &vartype1 eq C %then %do;
234 __zip1=put(&zipcode1, $5.);
235 %end;
236
237 %if &vartype2 eq N %then %do;
238 __zip2=put(&zipcode2,z5.);
239 %end;
240 %else %if &vartype2 eq C %then %do;
241 __zip2=put(&zipcode2, $5.);
242 %end;
243
244 %* Store zip code;
245 call symputx('z1', __zip1);
246 call symputx('z2', __zip2);
247
248 %* Some searches do not work unless you include city and date;
249 %* Example syntax for query: North+Olmsted,+OH+44070;
250 if not missing(zipcity(__zip1)) then
251 call symputx('zipcity1', tranwrd(catx(' ', compress(zipcity(__zip1), ","), __zip1), " ", "+"));
252 if not missing(zipcity(__zip2)) then
253 call symputx('zipcity2', tranwrd(catx(' ', compress(zipcity(__zip2), ","), __zip2), " ", "+"));
254
255 stop;
256 run;
257
258 %* If city not found, skip to next pair;
259 %if %length(&zipcity1)=0 or %length(&zipcity2)=0 %then
260 %put NOTE: At least one zip code not found in SASHELP.ZIPCODE, skipping to next pair.;
261
262 %if %length(&zipcity1) and %length(&zipcity2) %then %do;
263
264 filename src temp;
265
266 %* Before getting data, make sure the website exists;
267 filename headout TEMP;
268 proc http url="https://www.google.com/maps/dir/&zipcity1/&zipcity2/?force=lite"
269 headerout=headout;
270 run;
271
272 data _null_;
273 infile headout scanover truncover;
274 input @'HTTP/1.1' code 4. message $255.;
275 if _n_ = 1 then call symput("message", strip(message));
276 run;
277
278 %* If website exists then get data, otherwise put a message and continue;
279 %if "&message" eq "OK" %then %do;
280
281 proc http
282 url="https://www.google.com/maps/dir/&zipcity1/&zipcity2/?force=lite"
283 out=src;
284 run;
285
286 data _temp_ (keep = zip1 zip2 text--drive_duration);
287 infile src length = len lrecl = 32767;
288 input line $varying32767. len;
289 line = strip(line);
290 if index(line, 'mile') then indx = index(line, 'mile');
291 else if index(line, ' km') then indx = index(line, ' mi');
292 retain zip1 &z1 zip2 &z2;
293 if indx then do;
294 lineLen = length(line);
295 startPos = find(reverse(substr(line, 1, indx - 1)), '"');
296 endPos = 0;
297 do i=1 to 3 until(endPos = 0);
298 endPos = find(substr(line, indx, lineLen - indx), '"', endPos + 1);
299 end;
300 text = substr(line, indx - startPos, startPos + endPos);
301 drive_distance = input(compress(scan(text,1,'\'), '",', 'ai'), ?? best.);
302 drive_distance_units = compress(scan(text,1,'\'), , "kai");
303 drive_duration = compress(scan(text,3,'"'), "\");
304 format drive_distance comma32.2;
305 end;
306 if not missing(drive_distance) then output;
307 run;
308
309 %end;
310
311 %else %put NOTE: Website for &z1 and &z2 could not be accessed. Reason: &message.. ;
312
313 filename src clear;
314 filename headout clear;
315
316 %* add an observation to the data set _DISTANCE_;
317 proc append base=_distance_ data=_temp_;
318 run;
319
320 %end; %* end to if zipcity1 and zipcity2;
321
322 %* delete temp;
323 proc datasets lib=work nolist;
324 delete _temp_;
325 quit;

328 %let z2 = ;
329 %let zipcity1 = ;
330 %let zipcity2 = ;
331
332%end; %* end to j to nzips;

334%*****************************************************************************;
335%* ADD DISTANCE TO ORIGINAL DATA (ALLOW CHARACTER TO NUMERIC MERGE) ;
336%*****************************************************************************;
337proc sort data=_distance_ nodupkey;
338 by zip1 zip2;
339run;
340
341proc sql;
342 create table &outdata
343 as select a.*, b.drive_distance, b.drive_distance_units, b.drive_duration
344 %if %upcase(&straightline) in (1 YES Y) %then %do;
345 , zipcitydistance(a.&zipcode1, a.&zipcode2) as straight_line_distance_mi
346 %end;
347 from &data a left join _distance_ b
348 on %if &vartype1 eq N %then %do; put(a.&zipcode1, z5.) %end;
349 %else %if &vartype1 eq C %then %do; put(a.&zipcode1, $5.) %end;
350 =put(b.zip1, z5.) and
351 %if &vartype2 eq N %then %do; put(a.&zipcode2, z5.) %end;
352 %else %if &vartype2 eq C %then %do; put(a.&zipcode2, $5.) %end;
353 =put(b.zip2, z5.);
354quit;
355
356%* Additional tidy up;
357data &outdata (drop = drive_distance drive_distance_units drive_duration);
358 set &outdata;
359
360 **If same origin and destination put 0;
361 if
362 %if &vartype1 eq N %then %do; put(&zipcode1, z5.) %end;
363 %else %if &vartype1 eq C %then %do; put(&zipcode1, $5.) %end;
364 eq
365 %if &vartype2 eq N %then %do; put(&zipcode2, z5.) %end;
366 %else %if &vartype2 eq C %then %do; put(&zipcode2, $5.) %end;
367 then do;
368 drive_distance=0;
369 drive_distance_units='miles';
370 drive_duration='0 min';
371 end;
372
373 **convert km to miles;
374 if compress(drive_distance_units) in ("mile" "miles") then
375 drive_miles = drive_distance;
376 else if compress(drive_distance_units) = "km" then
377 drive_miles = round(divide(drive_distance, 1.609), 0.1);
378
379 **drive_time as numeric;
380 if find(upcase(drive_duration), 'DAY') then do;
381 if find(upcase(drive_duration), 'HR') and
382 find(upcase(drive_duration), 'MIN') then
383 drive_time = (input(scan(drive_duration, 1), 8.) * 86400) +
384 (input(scan(drive_duration, 3), 8.) * 3600) +
385 (input(scan(drive_duration, 5), 8.) * 60);
386 else if find(upcase(drive_duration), 'HR') and
387 not find(upcase(drive_duration), 'MIN') then
388 drive_time = (input(scan(drive_duration, 1), 8.) * 86400) +
389 (input(scan(drive_duration, 3), 8.) * 3600);
390 else if not find(upcase(drive_duration), 'HR') and
391 find(upcase(drive_duration), 'MIN') then
392 drive_time = (input(scan(drive_duration, 1), 8.) * 86400) +
393 (input(scan(drive_duration, 3), 8.) * 60);
394 else drive_time = input(scan(drive_duration, 1), 8.) * 86400;
395 end;
396
397 else if find(upcase(drive_duration), 'HR') then do;
398 if find(upcase(drive_duration), 'MIN') then
399 drive_time = (input(scan(drive_duration, 1), 8.) * 3600) +
400 (input(scan(drive_duration, 3), 8.) * 60);
401 else drive_time = input(scan(drive_duration, 1), 8.) * 3600;
402 end;
403
404 else if find(upcase(drive_duration), 'MIN') then do;
405 drive_time = input(scan(drive_duration, 1), 8.) * 60;
406 end;
407
408 format drive_miles comma32.2 drive_time time6.;
409 label drive_miles = "Driving distance (miles)"
410 drive_time = "Driving time (seconds, formated as hh:mm)"
411 straight_line_distance_mi = "Geodetic distance (miles)";
412run;
