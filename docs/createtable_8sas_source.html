1/**
2 @file
3 @brief Generates a simple report-ready table
4
5 @details
6 @par Purpose
7 Runs PROC REPORT and creates a simple report-ready table
8
9 @author Rocio Lopez
10
11 @date 02-01-2016
12
13 @version SAS 9.2 or later
14
15 @par Usage
16 A call to this macro has some or all of these parameters (defaults given):
17 @code{.sas}
18 %CreateTable(
19 ds =
20 ,title =
21 ,labelCol =
22 ,columns =
23 ,pValue =
24 ,pFmt = pvaluef.
25 ,footnote =
26 ,cWidth1 = 350
27 ,cWidth2 = 250
28 ,outputWidth =
29 ,fontSize = 11
30 ,fontStyle = Roman
31 ,fontFace= Times
32 ,boldedLabel = 0
33 ,boldedSigP = 1
34 ,sigP = 0.05
35 ,style = rtf
36 ,newPage = 0
37 ,page = portrait
38 ,rtfFile =
39 ,xmlFile =
40 ,pdfFile =
41 );
42 @endcode
43
44 @returns
45 A table produced by proc report
46
47 @note
48 @parblock
49 @li It is recommended that all variables be formatted and labeled
50
51 @li Unless running on SAS Studio, output might not look right in results
52 window but styling in RTF file will be fine
53
54 @li On occasion you might see an error message about the width of a column not
55 being between 1 and 78. This has to do with the listing destination being
56 open and does not affect the output. If you wish to save an error-free copy
57 of your log, use "ods listing close;" prior to calling the macro.
58 @endparblock
59
60 @param ds The data set to be used to generate table
61
62 @param labelCol Optional name of a single variable to be used as the label column;
63 column will be left justified
64
65 @param columns List of 1+ variable names to include in table; columns will be
66 center justified
67
68 @param title Optional title for table; should be surrounded by double quotes
69
70 @param pValue The numeric variable that contains the p-vlaues; column will be
71 center justified
72
73 @param pFmt Format for p-value column
74
75 @param footNote Optional footnote to be included under the table; should be
76 surrounded by double quotes
77
78 @param cWidth1 Specify width of label column in pixels, default is 350
79
80 @param cWidth2 Specify width of columns in pixels
81 @n Default is 300
82
83 @param outputWidth Specify output width of the table in terms of % of RTF
84 page; value should be 1-100 with 100 occupying 100% of the page
85 @n Default size is determined by PROC REPORT
86
87 @param fontSize Font size in pt
88 @n Default is 11
89
90 @param fontFace Specify font face
91 @n Default is Times
92
93 @param fontStyle Specify font style
94 @n Default is Roman
95
96 @param boldedLabel Specify whether label column uses bold font
97 @n Allowed options are:
98 @n 0 = Use normal font (DEFAULT)
99 @n 1 = Use bold font
100
101 @param boldedSigP Specify whether significant p-values use bold font
102 @n Allowed options are:
103 @n 0 = Use normal font
104 @n 1 = Use bold font (DEFAULT)
105
106 @param sigP Specify significance level for bold p-values
107 @n Default is 0.05
108
109 @param style ODS style to be used
110 @n Default is rtf
111
112 @param newPage Select to start new page; this only takes effect if inluding in
113 a larger ODS output file and not using the rtfFile, xlmFile or pdfFile options
114 within the macro
115 @n Allowed options are:
116 @n 0 = No, continue on same page
117 @n 1 = Yes, start new page (DEFAULT)
118
119 @param page Specify page orientation for ods. If Landscape used, user must
120 start new page in next section/figure/table call to revert to portrait
121@n Allowed options are:
122@n portrait (DEFAULT)
123@n landscape
124
125 @param rtffile Optional RTF name in which to store the output, it should be
126 surrounded in quotes, e.g. "~userid/consult/out.rtf"
127
128 @param xmlfile Optional Excel XML file name in which to store the output, it
129 should be surrounded in quotes, e.g. "~userid/consult/out.xml"
130 @n This has some stylying issues and is not recommended
131
132 @param pdffile Optional PDF file name in which to store the output, it should
133 be surrounded in quotes, e.g. "~userid/consult/out.pdf"
134
135 @par Example(s)
136 @code{.sas}
137 proc reg data=sashelp.class;
138 model weight = height age;
139 ods output parameterEstimates=pe;
140 run;
141
142 %CreateTable(
143 ds = pe,
144 title = "Example",
145 labelCol = Variable,
146 columns = estimate stdErr,
147 pValue = probT,
148 footnote = "My footnote here");
149 @endcode
150

158******************************************************************************;
159**Start macro ;
160******************************************************************************;
161%macro createTable(
162 Ds = ,
163 Title = ,
164 LabelCol = ,
165 Columns = ,
166 pvalue = ,
167 pFmt = pvaluef.,
168 Footnote = ,
169 CWidth1 = 350,
170 CWidth2 = 250,
171 OutputWidth = ,
172 fontsize = 11,
173 fontstyle = Roman,
174 fontface= Times,
175 boldedlabel = 0,
176 boldedsigp = 1,
177 sigp = 0.05,
178 style = rtf,
179 NewPage = 0,
180 Page = portrait,
181 rtffile = ,
182 xmlfile = ,
183 pdffile =
184 );
185
186******************************************************************************;
187**Local macro variables ;
188******************************************************************************;
189%LOCAL
190 DS TITLE LABELCOL COLUMNS PVALUE FOOTNOTE CWIDTH NEWPAGE STPAGE PAGE ERRORFLAG;
191
192******************************************************************************;
193**Error checks ;
194******************************************************************************;
195%LET ErrorFlag=0;
196
197****************;
198**Verify dataset;
199****************;
200/**Check if data set specified in call **/
201%IF (&Ds= ) %THEN %DO;
202 %PUT ERROR: No input data set identified in DATA parameter.
203 Macro will stop executing.;
204 %LET ErrorFlag=1;
205%END;
206
207/** Check if data set exists**/
208%IF %SYSFUNC(exist(&Ds)) ne 1 %THEN %DO;
209 %PUT ERROR: Data set &Ds does not exist. Macro will stop executing;
210 %LET ErrorFlag=1;
211%END;
212
213***************************************************;
214**Verify all other required parameters are provided;
215***************************************************;
216%IF (&Columns= ) %THEN %DO;
217 %PUT ERROR: Required parameter COLUMNS not provided. Macro will stop executing.;
218 %LET ErrorFlag=1;
219%END;
220
221********************************;
222**Check use of "" where required;
223********************************;
224%IF %length(&Title)>0 and %SYSEVALF(%SYSFUNC(indexc(&Title., '"'))^= 1) %THEN %DO;
225 %PUT
226 ERROR: Double quotes are required for TITLE. Macro will stop executing.;
227 %LET ErrorFlag=1;
228%END;
229
230%IF %length(&Footnote)>0 and %SYSEVALF(%SYSFUNC(indexc(&Footnote., '"'))^= 1) %THEN %DO;
231 %PUT
232 ERROR: Double quotes are required for FOOTNOTE. Macro will stop executing.;
233 %LET ErrorFlag=1;
234%END;
235
236%IF %length(&RTFFILE)>0 and %SYSEVALF(%SYSFUNC(indexc(&RTFFILE., '"'))^= 1) %THEN %DO;
237 %PUT
238 ERROR: Double quotes are required for RTFFILE. Macro will stop executing.;
239 %LET ErrorFlag=1;
240%END;
241
242******************************************************************************;
243**If no errors, start macros ;
244******************************************************************************;
245%IF &ErrorFlag=0 %THEN %DO;
246
247***************;
248**Define Format;
249***************;
250proc format;
251 picture Pvaluef (round)
252 0.985 - high = "0.99" (NoEdit)
253 0.10 -< 0.985 = "9.99"
254 0.001 -< 0.10 = "9.999"
255 0 -< 0.001 = "<0.001" (NoEdit)
256 . = " ";
257run;
258
259***********************************;
260**Set macro variable for start page;
261***********************************;
262%IF &NEWPAGE=1 %THEN %LET STPAGE=YES;
263%ELSE %LET STPAGE=NO;
264
265************************************************************;
266**Orientation of page (start new page even if not requested);
267**New Page ;
268************************************************************;
269options nodate nonumber;
270%IF &NewPage.=1 or %UPCASE(&Page.)=LANDSCAPE %THEN %DO;
271 options orientation=&Page.;
272 ods rtf startpage=NOW;
273%END;
274
275*********************************;
276**Open requested ODS destinations;
277*********************************;
278%IF (&RTFFILE^= ) %THEN %DO;
279 ods rtf file=&RTFFILE style=&style startpage=&STPAGE;
280%END;
281
282%IF (&XMLFILE^=) %THEN %DO;
283 ods tagsets.ExcelXP file=&XMLFILE style=&style;
284%END;
285
286%IF (&PDFFILE^= ) %THEN %DO;
287 ods pdf file=&PDFFILE style=&style bookmarkgen=no bookmarklist=none startpage=&STPAGE;
288%END;
289
290***************************;
291**Count # of columns needed;
292***************************;
293%let NumCols=%sysfunc(countw(&Columns));
294
295***********;
296**Add Table;
297***********;
298proc report data=&Ds nowd
299 style(report)={bordercolor=black cellpadding=3
300 %if (&outputwidth^= ) %then %do; outputwidth=&outputwidth.% %end;
301 font_size=&fontsize.pt font_face=&fontface. FONTSTYLE=&fontstyle.}
302
303 style(lines)={background=white foreground=black
304 font_size=9pt font_face=&fontface. FONTSTYLE=&fontstyle.
305 protectspecialchars=off}
306
307 style(column)={background=white foreground=black bordercolor=black
308 font_size=&fontsize.pt font_face=&fontface. FONTSTYLE=&fontstyle.
309 font_weight=medium}
310
311 style(header)={background=white foreground=black bordercolor=black
312 font_weight=bold FONTSTYLE=&fontstyle.
313 font_size=&fontsize.pt font_face=&fontface.};
314
315 column (%IF (&LabelCol.^= ) %THEN %DO; &LabelCol. %END;
316 &Columns.
317 %IF (&pvalue.^= ) %THEN %DO; &pvalue. _p %END;
318 );
319
320 /** Title **/
321 %IF (&Title.^= ) %THEN %DO;
322 compute before _PAGE_ /style = {font_weight=bold font_size=&fontsize.pt
323 just=left borderbottomwidth=3
324 borderbottomcolor=black bordertopcolor=white
325 borderleftcolor=white borderrightcolor=white};
326 line &Title.;
327 endcomp;
328 %END;
329
330 /** Variable name column **/
331 %IF (&LabelCol.^= ) %THEN %DO;
332 define &LabelCol/order=data left
333 style(header)={just=left}
334 style(column)={cellwidth=&cwidth1.
335 %if &boldedlabel=1 %then %do; font_weight=bold %end;
336 };
337 %END;
338
339 /** Loop through columns **/
340 %DO Loop=1 %TO &NumCols.;
341 define %QSCAN(&Columns, &Loop.)
342 / display
343 style(column)={just = center vjust = center cellwidth = &cwidth2.
344 protectspecialchars=off};
345 %END;
346
347 /** p-value column **/
348 %IF (&pvalue.^= ) %THEN %DO;
349 define &pvalue/display noprint;
350 define _p/computed "p-value"
351 format=&pfmt.
352 style(header)={just=center}
353 style(column)={just=center vjust = center cellwidth=85};
354 compute _p;
355 _p=&pvalue;
356 %if &boldedsigp=1 %then %do;
357 if _p ne . and _p < &sigp. then call define(_col_,
358 "style", "style=[font_weight=bold fontstyle=italic]");
359 %end;
360 endcomp;
361 %END;
362
363 /**Add Footnote**/
364 %IF (&Footnote.^= ) %THEN %DO;
365 compute after/style(lines)={font_size=9pt
366 just=left bordertopwidth=3
367 borderbottomcolor=white bordertopcolor=black
368 borderleftcolor=white borderrightcolor=white};
369 line &Footnote.;
370 endcomp;
371 %END;
372
373run;
374************************;
375**Close ODS destinations;
376************************;
377%IF (&RTFFILE^= ) %THEN %DO;
378 ods rtf close;
379%END;
380
381%IF (&XMLFILE^=) %THEN %DO;
382 ods tagsets.ExcelXP close;
383%END;
384
385%IF (&PDFFILE^= ) %THEN %DO;
386 ods pdf close;
387%END;
388
389******************************************************************************;
390**END if errorflag=0 loop ;
391******************************************************************************;
392%END;
393
394******************************************************************************;
395**Close macros definition ;
396******************************************************************************;
397%MEND createTable;
