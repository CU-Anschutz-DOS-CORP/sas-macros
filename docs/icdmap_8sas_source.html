1/**
2 @file
3 @brief Maps ICD-9 to ICD-10 and vice-versa
4
5 @details
6 @par Purpose
7 This macro uses CMS GEMs and ICD code lists to map ICD-9 to ICD-10
8 and ICD-10 to ICD-9 diagnosis or procedure codes.
9 It can also be used to generate a list of select codes with
10 descriptions and can also provide a list of code changes
11 for ICD-10 codes.
12
13 @author Rocio Lopez
14
15 @date 12-10-2020
16
17 @version SAS 9.2 or later
18
19 @par Usage
20 A call to this macro has some or all of these parameters (defaults given):
21 @code{.sas}
22 %icdmap(
23 libname =
24 ,icd9 =
25 ,icd10 =
26 ,codetype = dx
27 ,map = 1
28 ,checkconversion10 = 0
29 );
30 @endcode
31
32 @returns
33 One or more tables printed to the results window.
34
35 @param libName Library name assigned to the directory where the CMS
36 tables are stored
37
38 @param icd9 List of ICD-9 codes to map, each enclosed by ''
39 @n This expression will be used with the IN () operator
40
41 @param icd10 List of ICD-10 codes to map, each enclosed by ''
42 @n This expression will be used with the IN () operator
43
44 @param codeType Type of codes provided
45 @n Valid options are: DX for diagnosis (default) or PRC for
46 procedures
47
48 @param map Logical for whether or not to map across versions
49 @n Valid options are: 0 for No, 1 for Yes (default)
50
51 @param checkConversion10 If ICD-10 list given, print a list of
52 changes in specified codes, if any
53 @n Valid options are: 0 for No, 1 for Yes (default)
54
55 @note
56 @parblock
57 @li User must have necessary CMS tables stored as .sas7bdat files
58 and connect to the location using LIBNAME statement. Go to
59 https://github.com/lopezr/public-files/blob/main/icdMap/CMS%20ICD%20Tables%20Used%20by%20%25icdMap%20SAS%20macro.pdf
60 to see a list of the tables we use. You may need to update the
61 table and/or column names within the macro depending on your set-up.
62
63 @li You can omit the periods from the codes.
64
65 @li If you have a full range of codes to use (456.xx, for example),
66 you can query the tables prior to using the macro and create a
67 macro list containing the codes. See Example 3 below.
68 @endparblock
69
70 @par Example(s)
71 @code{.sas}
72 **Example 1 - Mapping ICD-9-CM to ICD-10-CM ;
73 %icdmap(
74 libname = cms
75 ,icd9 = "456.0" "456.20"
76 );
77
78 **Example 2 - Mapping ICD-9-CM to ICD-10-CM and vice vers;
79 %icdmap(
80 libname = cms
81 ,icd9 = "456.0" "456.20"
82 ,icd10 = "I85.01" "I85.11"
83 ) ;
84
85 **Example 3 - Creating a macro variable with list of codes corresponding to a range ;
86 *Create list of all 456.xx ICD-9-CM codes;
87 proc sql noprint;
88 select compress(ICD9_CODE, " .")
89 into :list456 separated by '" "'
90 from cms.CMS_ICD9_DX_DESC_2014
91 where substr(ICD9_CODE, 1, 3) eq "456"
92 order by ICD9_CODE;
93 quit ;
94 %put ***LIST: "&list456"***;
95
96 *Map ICD-9-CM 456.xx to ICD-10-CM" ;
97 %icdmap(
98 libname = cms
99 ,icd9 = "&list456") ;
100 @endcode
101
102 @par Revision History
103 @b 05-12-2023 Updated name (version) of conversion table
104 @n @b 10-18-2023 Turned off printing of description when only the the conversion is requested
105**/
106
107%macro icdmap(
108 libname = ,
109 icd9 = ,
110 icd10 = ,
111 codetype = dx,
112 map = 1,
113 checkconversion10 = 0) / minoperator;
114
115%put ************************************************************;
116%put NOTE: Macro &sysmacroname is starting to run. ;
117%put ************************************************************;
118
119%local ERRORFLAG DX9TO10 DX10TO9 DXDESC9 DXDESC10 PRC9TO10 PRC10TO9
120PRCDESC9 PRCDESC10 LAB9 LAB10 SUFFIX9 SUFFIX10;
121
122%put ************************************************************;
123%put ************************************************************;
124%put * ERROR CHECKING *;
125%put ************************************************************;
126%put ************************************************************;
127
128%let errorflag = 0 ;
129
130%put *************************************************;
131%put * CHECKING LIBNAME GIVEN AND EXISTS *;
132%put *************************************************;
133
134%if %length(&libname) eq 0 %then %do ;
135 %put ERROR: Parameter LIBNAME must be provided. Macro will stop executing. ;
136 %let errorflag = 1 ;
137%end ;
138
139%else %if %length(&libname) gt 0 %then %do ;
140 %if %sysfunc(libref(&libname)) gt 0 %then %do ;
141 %put ERROR: Library %upcase(&libname) does not exist. Macro will stop executing. ;
142 %let errorflag = 1 ;
143 %end ;
144 %if %sysfunc(libref(&libname)) lt 0 %then %do ;
145 %put WARNING: Library %upcase(&libname) exists, but the pathname is in question. Macro will continue but this might cause errors. ;
146 %end ;
147%end ;
148
149%put *************************************************;
150%put * CHECKING AT LEAST 1 CODE LIST PROVIDED *;
151%put *************************************************;
152
153%if %length(&icd9) eq 0 and %length(&icd10) eq 0 %then %do ;
154 %put ERROR: Either ICD9LIST or ICD10LIST must be provided. Macro will stop executing. ;
155 %let errorflag = 1 ;
156%end ;
157
158%put *************************************************;
159%put * CHECKING CODETYPE PARAMETER *;
160%put *************************************************;
161
162%if not(%upcase(&codetype) in (DX PRC)) %then %do ;
163 %put ERROR: CODETYPE must be either DX or PRC. ;
164 %put CODETYPE = **&codetype** ;
165 %let errorflag = 1 ;
166%end ;
167
168%put *************************************************;
169%put * CHECKING MAP PARAMETER *;
170%put *************************************************;
171
172%if not(%upcase(&map) in (0 1 NO YES)) %then %do ;
173 %put WARNING: MAP must be either 0, 1, no or yes. ;
174 %put MAP = **&map** ;
175 %put Default value of 1 (YES) will be used instead. ;
176 %let map = 1 ;
177%end;
178
179%put *************************************************;
180%put * CHECKING CHECKCONVERSION10 PARAMETER *;
181%put *************************************************;
182
183%if not(%upcase(&checkconversion10) in (0 1 NO YES)) %then %do ;
184 %put WARNING: CHECKCONVERSION10 must be either 0, 1, no or yes. ;
185 %put CHECKCONVERSION10 = **&checkconversion10** ;
186 %put Default value of 1 (YES) will be used instead. ;
187 %let map = 1 ;
188%end;
189
190%if &errorflag %then %do ;
191 %put *************************************************;
192 %put * ABORT EXECUTION BECAUSE OF ERROR *;
193 %put *************************************************;
194 data _null_ ;
195 abort 3 ;
196 run ;
197%end ;
198
199%put *************************************************;
200%put * PROCEEDING WITH MACRO, NO ERRORS FOUND *;
201%put *************************************************;
202
203%put ************************************************************;
204%put ************************************************************;
205%put * DEFINE TABLES AND NAMES *;
206%put ************************************************************;
207%put ************************************************************;
208
209%* If you want to use different versions, change the names here but be mindful that column names match ;
210%let codetype = %upcase(&codetype);
211
212%* CPRH Teradata Data Lab Table names ;
213%let map9to10 = &libname..CMS_GEM_&codetype._ICD9_ICD10_2018 ;
214%let map10to9 = &libname..CMS_GEM_&codetype._ICD10_ICD9_2018 ;
215%let desc10 = &libname..CMS_ICD10_&codetype._DESC_2018 ;
216%let desc9 = &libname..CMS_ICD9_&codetype._DESC_2014 ;
217%let conversion = &libname..CMS_ICD10_&codetype._CONVERSION_2023 ;
218
219%* Misc set-up for column names and title labels ;
220%if %upcase(&codetype) eq DX %then %do;
221 %let lab9 = ICD9 ;
222 %let lab10 = ICD10 ;
223 %let suffix9 = CM diagnosis;
224 %let suffix10 = CM ;
225%end;
226
227%else %if %upcase(&codetype) eq PRC %then %do;
228 %let lab9 = PROCEDURE ;
229 %let lab10 = PROCEDURE ;
230 %let suffix9 = CM procedure;
231 %let suffix10 = PCS ;
232%end;
233
234%let icd9 = %upcase(%sysfunc(compress(&icd9, .)));
235%let icd10 = %upcase(%sysfunc(compress(&icd10, .)));
236
237%if %length(&icd9) %then %do;
238
239 %put ************************************************************;
240 %put ************************************************************;
241 %put * MANAGE ICD9 CODE LIST *;
242 %put ************************************************************;
243 %put ************************************************************;
244
245 %* List code with description ;
246 %if %upcase(&map) in (0 NO) %then %do ;
247 title3 "Description of ICD-9-&suffix9 codes" ;
248 proc sql ;
249 select *
250 from &desc9
251 where compress(&lab9._CODE, ". ") in (&icd9)
252 order by &lab9._CODE ;
253 quit ;
254 title3 ;
255 %end ;
256
257 %* Map 9 to 10 ;
258 %else %do ;
259 title3 "ICD-9-&suffix9 to ICD-10-&suffix10 mapping" ;
260 proc sql ;
261 select
262 %if %upcase(&codetype) eq DX %then %do ;
263 a.ICD9_Code,
264 %end ;
265 %else %if %upcase(&codetype) eq PRC %then %do ;
266 compress(substr(a.ICD9_Code,1,2))||"."||compress(substr(a.ICD9_Code,3,2)) as ICD9_code,
267 %end ;
268 b.Long_Description as ICD9_Description label = "ICD9_Description",
269 a.ICD10_Code, c.Long_Description as ICD10_Description label = "ICD10_Description",
270 a.Flag as MappingFlags label = "Mapping Flags"
271 /* , a.ApproximateFlag, a.NoMapFlag, a.CombinationFlag, a.ScenarioFlag, a.ChoiceListFlag */
272 %if %length(&icd10) gt 0 %then %do ;
273 ,
274 case
275 when compress(a.ICD10_CODE, ". ") not in (&icd10) then "1"
276 else " "
277 end as NotInICD10List
278 %end ;
279 from &map9to10 (where = (compress(ICD9_CODE, ". ") in (&icd9))) a
280 left join &desc9 b
281 on a.ICD9_CODE = b.&lab9._Code
282 left join &desc10 c
283 on a.ICD10_CODE = c.&lab10._Code
284 order by a.ICD9_CODE, a.Flag ;
285 quit ;
286 title3 ;
287 %end ;
288
289%end ; %* end to icd9 loop ;
290
291%if %length(&icd10) %then %do;
292
293 %put ************************************************************;
294 %put ************************************************************;
295 %put * MANAGE ICD10 CODE LIST *;
296 %put ************************************************************;
297 %put ************************************************************;
298
299 %* List code with description ;
300 %if %upcase(&map) in (0 NO) and %upcase(&checkconversion10) in (0 NO) %then %do ;
301 title3 "Description of ICD-10-&suffix10 codes" ;
302 proc sql;
303 select
304 &lab10._CODE as ICD10_Code label = "ICD10_Code",
305 /*%if %upcase(&codetype) eq DX %then %do; SHORT_DESCRIPTION, %end ; */
306 LONG_DESCRIPTION
307 from &desc10
308 where compress(&lab10._CODE, ". ") in (&icd10)
309 order by &lab10._CODE;;
310 quit;
311 title3 ;
312 %end ;
313
314 %* Map 10 to 9 ;
315 %if %upcase(&map) in (1 YES) %then %do;
316 title3 "ICD-10-&suffix10 to ICD-9-&suffix9 mapping" ;
317 proc sql;
318 select
319 a.ICD10_Code, c.Long_Description as ICD10_Description label = "ICD10_Description",
320 %if %upcase(&codetype) eq DX %then %do ;
321 a.ICD9_Code,
322 %end ;
323 %else %if %upcase(&codetype) eq PRC %then %do ;
324 compress(substr(a.ICD9_Code,1,2))||"."||compress(substr(a.ICD9_Code,3,2)) as ICD9_code,
325 %end ;
326 b.Long_Description as ICD9_Description label = "ICD9_Description",
327 a.Flag as MappingFlags label = "Mapping Flags"
328 /* , a.ApproximateFlag, a.NoMapFlag, a.CombinationFlag, a.ScenarioFlag, a.ChoiceListFlag */
329 %if %length(&icd9) gt 0 %then %do ;
330 ,
331 case
332 when compress(a.ICD9_CODE, ". ") not in (&icd9) then "1"
333 else " "
334 end as NotInICD9List
335 %end ;
336 from &map10to9 (where = (compress(ICD10_CODE, ". ") in (&icd10))) a
337 left join &desc9 b
338 on a.ICD9_CODE = b.&lab9._Code
339 left join &desc10 c
340 on a.ICD10_CODE = c.&lab10._Code
341 order by a.ICD10_CODE, a.Flag ;
342 quit;
343 title3 ;
344 %end ;
345
346 %* Changes in codes ;
347 %if %upcase(&checkconversion10) in (1 YES) %then %do ;
348 title3 "Checking for newer or older ICD-10-&suffix10 codes" ;
349 proc sql ;
350 select *
351 from &conversion
352 where
353 %if %upcase(&codetype) eq DX %then %do ;
354 compress(CURRENT_ICD10_CODE, ". ") in (&icd10) or compress(OLD_ICD10_CODE, ". ") in (&icd10)
355 %end;
356 %else %if %upcase(&codetype) eq PRC %then %do ;
357 compress(CURRENT_ICD10_CODE, ". ") in (&icd10) or compress(OLD_ICD10_CODE, ". ") in (&icd10)
358 %end;
359 order by
360 %if %upcase(&codetype) eq DX %then %do ; CURRENT_ICD10_CODE, OLD_ICD10_CODE %end ;
361 %else %if %upcase(&codetype) eq PRC %then %do ; CURRENT_ICD10_CODE, OLD_ICD10_CODE %end ; ;
362 quit;
363 title3 ;
364 %end ;
365%end ; %* end to icd10 loop ;
366
367%put ************************************************************;
368%put NOTE: Macro &sysmacroname has finished. ;
369%put ************************************************************;
370
371%mend icdmap ;
