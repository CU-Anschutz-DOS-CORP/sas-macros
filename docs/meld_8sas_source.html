2 @file
3 @brief Calculates MELD, MELD-Na, and MELD 3.0
4
5 @details
6 @par Purpose
7 This macro runs within a DATA step and calculates Model For End-Stage Liver
8 Disease (MELD), MELD-Na, and MELD 3.0 scores
9
10 @author Rocio Lopez
11
12 @date 07-14-2023
13
14 @version SAS 9.4 or later
15
16 @par Usage
17 A call to this macro has all or some of these parameters:
18 @code{.sas}
19 %meld(
20 ageAtListing =
21 ,female =
22 ,bilirubin =
23 ,sodium =
24 ,inr =
25 ,creatinine =
26 ,albumin =
27 ,dialysis =
28 ,prefix =
29 ,cap = 1
30 );
31 @endcode
32
33 @returns
34 The following columns will be added to the data set being created:
35 meld, meldNa, meld3
36
37 @param ageAtListing Variable containing age at time of liver transplant listing
38
39 @param female Variable indicating whether the candidate is female
40 @n This must be coded as 1 = female and 0 = male
41
42 @param bilirubin Variable containing bilirubin in mg/dL
43
44 @param sodium Variable containing serum sodium in mEq/L
45
46 @param inr Variable containing INR
47
48 @param creatinine Variable containing serum creatinine in mg/dL
49
50 @param albumin Variable containing albumin in g/dL
51
52 @param dialysis Variable indicating whether the candidate received two or more
53 dialysis treatments within 7 days prior to the serum creatinine test or received
54 24 hours of continuous veno-venous hemodialysis (CVVHD) prior to the serum creatinine test
55 @n This must be coded as 1 = yes and 0 = no
56
57 @param prefix Optional prefix for meld score variable names
58 @n For example, if prefix = r_ then the 3 variables created will be r_meld r_meldNa r_meld3
59
60 @param cap Requests that the scores be capped between 6-40 as per OPTN policy
61 @n Allowed options are:
62 @n 0 = No, calcuate actual score without capping
63 @n 1 = Yes, score will be capped betwee 6 and 40 (DEFAULT)
64
65 @note
66 @parblock
67 @li This macro will only run from within a data step.
68
69 @li If dialysis is missing, calculations will assume no dialysis.
70
71 @li OPTN policy used MELD (meld) prior to 01/2016, MELD-Na (meldNa) from 01/??/2016 to 07/12/2023,
72 and MELD 3.0 starting on 7/13/2023
73
74 @li Notice of OPTN Policy Change (MELD-Na to MELD 3.0:

76
77 @li Reference for current policy using MELD 3.0 (as of 7/13/2023)

114 20 0 2.5 131 1.0 1.2 3.8 0

140%macro meld(
141 ageAtListing =
142 ,female =
143 ,bilirubin =
144 ,sodium =
145 ,inr =
146 ,creatinine =
147 ,albumin =
148 ,dialysis =
149 ,prefix =
150 ,cap = 1
151);
152
153 informat biliForMeld inrForMeld scrForMeld scrForMeld3 naForMeld albForMeld
154 &prefix.meld &prefix.meldNa &prefix.meld3 best.;
155
156 if nmiss(&bilirubin, &sodium, &inr, &creatinine) = 0 then do;
157 /* Set bounds as per policy */
158 biliForMeld = max(&bilirubin, 1);
159 inrForMELD = max(&inr, 1);
160
161 if &creatinine > 4 or &dialysis = 1 then scrForMeld = 4;
162 else scrForMeld = max(&creatinine, 1);
163
164 naForMeld = min(max(&sodium, 125), 137);
165
166 /* Calculate MELD */
167 &prefix.meld =
168 round(
169 (0.957 * log(scrForMeld)) +
170 (0.378 * log(biliForMeld)) +
171 (1.120 * log(inrForMeld)) +
172 0.643
173 , 0.1) * 10;
174
175 /* Calcute MELD-Na */
176 if &prefix.meld gt 11 then &prefix.meldNa =
177 round(
178 &prefix.meld +
179 (1.32 * (137 - naForMeld)) -
180 (0.033 * &prefix.meld * (137 - naForMeld))
181 , 1);
182 else &prefix.meldNa = &prefix.meld;
183 end; **end of meld, meldna calc;
184
185 if nmiss(&ageAtListing, &female, &bilirubin, &sodium, &inr, &creatinine, &albumin) = 0 then do;
186 /* Set SCr and albumin bounds as per policy */
187 if &creatinine > 3 or &dialysis = 1 then scrForMeld3 = 3;
188 else scrForMeld3 = max(&creatinine, 1);
189
190 albForMeld = min(max(&albumin, 1.5), 3.5);
191
192 /* Calculate MELD 3.0 */
193 &prefix.meld3 =
194 round(
195 (1.33 * &female * (&ageAtListing ge 18)) +
196 (4.56 * log(biliForMeld)) +
197 (0.82 * (137 - naForMeld)) -
198 (0.24 * (137 - naForMeld) * log(biliForMeld)) +
199 (9.09 * log(inrForMeld)) +
200 (11.14 * log(scrForMeld3)) +
201 (1.85 * (3.5 - albForMeld)) -
202 (1.83 * (3.5 - albForMeld) * log(scrForMeld3)) -
203 (1.33 * (&ageAtListing ge 18)) +
204 7.33,
205 1);
206 end; **end of meld3 calc;
207
208 %if &cap = 1 %then %do;
209 if &prefix.meld lt 6 then &prefix.meld = 6;
210 else if &prefix.meld gt 40 then &prefix.meld = 40;
211
212 if &prefix.meldNa lt 6 then &prefix.meldNa = 6;
213 else if &prefix.meldNa gt 40 then &prefix.meldNa = 40;
214
215 if &prefix.meld3 lt 6 then &prefix.meld3 = 6;
216 else if &prefix.meld3 gt 40 then &prefix.meld3 = 40;
217
218 %end;
219
220 drop biliForMeld inrForMeld scrForMeld scrForMeld3 naForMeld albForMeld;
221
222%mend meld;
