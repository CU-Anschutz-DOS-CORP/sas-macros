57%macro getNhanesData(cycle = , files = , savein = );
58
59/****************************************************************************************************
60 * READ IN AND SAVE XPT DATA FILEs
61 ****************************************************************************************************/
62%* Count number of files;
63%let NumFiles=0;
64%do %while(%scan(&files,&NumFiles+1) ^= ); %let NumFiles=%eval(&NumFiles+1); %end;
65
66%* Loop through each file and get formats;
67%do i=1 %to &NumFiles;
68
69 %* Get current file;
70 %let file=%upcase(%scan(&files, &i));
71
72 %* Before getting data, make sure the website exists;
73 filename headout TEMP;
74 proc http url="https://wwwn.cdc.gov/Nchs/Nhanes/&cycle/&file..XPT"
75 headerout=headout;
76 run;
77
78 data _null_;
79 infile headout scanover truncover;
80 input @'HTTP/1.1' code 4. message $255.;
81 if _n_ = 1 then call symput("message&i", strip(message));
82 run;
83
84 %* If file exists then get, otherwise put a message and continue;
85 %if "&&message&i" eq "OK" %then %do;
86
87 %* detect proper delim for UNIX vs. Windows;
88 %let delim=%sysfunc(ifc(%eval(&sysscp. = WIN),\,/));
89
90 %* create a name for our downloaded ZIP;
91 %let fileloc = &savein.&delim.&file..XPT;
92 filename download "&fileloc";
93
94 %* Download the file from the Internet;
95 proc http
96 method='GET'
97 url="https://wwwn.cdc.gov/Nchs/Nhanes/&cycle/&file..XPT"
98 out=download;
99 run;
100
101 %end;
102
103 %else %put WARNING: &file could not be downloaded. Reason: &&message&i... ;
104
105%end;
106
107%mend getNhanesData;
