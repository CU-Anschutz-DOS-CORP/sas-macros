1/**
2 @file
3 @brief Creates a format catalog from NHANES documentation
4
5 @details
6 @par Purpose
7 This macro will read NHANES HTML documentation pages and create a format
8 catalog and/or code file for the data
9
10 @author Rocio Lopez
11
12 @date 09-18-2018
13
14 @version SAS 9.4 or later
15
16 @par Usage
17 A call to this macro has some or all of these parameters (defaults given):
18 @code{.sas}
19 %nhanesFmtScrape(
20 cycle =
21 ,files =
22 ,omitmissing = 1
23 ,fmtlibrary = work
24 ,fmtcatalog = nhanesFormats
25 ,fmtcodefile =
26 ,prints = 1
27 );
28 @endcode
29
30 @returns
31 SAS format catalog and/or .SAS file with PROC FORMAT code
32
33 @note
34 @parblock
35 @li Each format will be named VarNameF.
36
37 @li It is possible for format labels to have % or & in which case a
38 Warning message will be seen in the log (e.g. WARNING: Apparent
39 symbolic reference LT not resolved.); this can be ignored.
40
41 @endparblock
42
43 @param cycle NAHES cycle for html file(s), format should be YYYY-YYYY
44
45 @param files List of 1 or more NHANES data files for which to extract formats
46
47 @param omitMissing If yes, will omit formats for missing values
48 @n Allowed options are:
49 @n 1, yes, y for omitting missing values (DEFAULT)
50 @n 0, no, n for including missing values
51
52 @param fmtLibrary SAS library in which to store the format catalog
53 @n Default is WORK
54
55 @param fmtCatalog Name of generated SAS format catalog
56 @ Default is nhanesFormats
57
58 @param fmtCodeFile Optional name of .SAS file in which to save PROC FORMAT code
59
60 @param prints If yes, prints a summary of all formats in catalog
61 @n Allowed are are:
62 @n Allowed options are:
63 @n 1, yes, y to print catalog (DEFAULT)
64 @n 0, no, n to not print catalog
65
66 @sa getNhanesData.sas, imptallxpt.sas
67
68 @par Example(s)
69 @code{.sas}
70 libname fmt "./nhanes2015to16";
71 %NHANESFmtScrape(
72 cycle = 2015-2016
73 ,files = DEMO_I BMX_I
74 ,fmtLibrary = fmt
75 ,fmtCodeFile = "./nhanes2015to16/formats.sas"
76 );
77 @endcode
78
79 @par Revision History
80 @b 03-25-2021 Changes to allow for $format
81 @b 02-13-2020 Added option to omit formats for missing values
82**/

84%macro nhanesFmtScrape(
85 cycle = ,
86 files = ,
87 omitmissing = 1,
88 fmtlibrary = work,
89 fmtcatalog = nhanesFormats,
90 fmtcodefile = ,
91 prints = 1);
92
93%*****************************************************************************;
94%* ERROR CHECKING ;
95%*****************************************************************************;
96%let errorflag=0;
97options minoperator;
98
99%* Required parameters provided;
100%if %length(&cycle) = 0 %then %do;
101 %put ERROR: Required parameter CYCLE is not provided. Macro will stop executing.;
102 %let errorflag=1;
103%end; **end to if length(cycle) = 0 ;
104
105%if %length(&files) = 0 %then %do;
106 %put ERROR: Required parameter FILES is not provided. Macro will stop executing.;
107 %let errorflag=1;
108%end; **end to if length(cycle) = 0 ;
109
110%* Check if requested library is assigned;
111%if (%sysfunc(libref(&fmtlibrary))) ne 0 %then %do;
112 %put ERROR: Library %upcase(&fmtlibrary) has not been assigned. Macro will stop executing.;
113 %let errorflag=1;
114%end;
115
116%* PrintS is 0 1 y n yes no;
117%if not(%upcase(&prints) in (0 1 YES NO Y N)) %then %do;
118 %put WARNING: PRINTS must be either 0, 1, no, yes, n or y.;
119 %put PRINTS = **&prints**;
120 %put Default value of YES will be used instead.;
121 %let prints=1;
122%end; **end to if prints not in ();
123
124%* OmitMissing is 0 1 y n yes no;
125%if not(%upcase(&omitmissing) in (0 1 YES NO Y N)) %then %do;
126 %put WARNING: OMITMISSING must be either 0, 1, no, yes, n or y.;
127 %put OMITMISSING = **&prints**;
128 %put Default value of YES will be used instead.;
129 %let omitmissing=1;
130%end; **end to if omitmissing not in ();
131
132%*****************************************************************************;
133%* IF ANY ERRORS FOUND THEN ABORT EXECUTION ;
134%*****************************************************************************;
135%if &errorflag eq 1 %then %do;
136 data _null_;
137 abort 3;
138 run;
139%end;
140
141%*****************************************************************************;
142%* IF NO ERRORS, START MACRO ;
143%*****************************************************************************;
144
145%* Count number of files;
146%let NumFiles=0;
147%do %while(%scan(&files,&NumFiles+1) ^= ); %let NumFiles=%eval(&NumFiles+1); %end;
148
149%* Loop through each file and get formats;
150%do i=1 %to &NumFiles;
151
152 %* Get current file;
153 %let file=%scan(&files, &i);
154
155 %* Fetch webpage ;
156 filename src temp;
157 proc http
158 method="GET"
159 url="https://wwwn.cdc.gov/Nchs/Nhanes/&cycle/&file..htm"
160 out=src;
161 run;
162
163 %* Read the entire file and skip the blank lines ;
164 %* the LEN indicator tells us the length of each line ;
165 data doc;
166 infile src length=len lrecl=99999;
167 input line $varying32767. len;
168 line = strip(line);
169 if len>0;
170 run;
171
172 %* Parse the lines and keep codebook info ;
173 data codebook (keep=Name Value Code);
174 length Name Value Code $ 32767;
175 set doc;
176
177 %* Get Variable Names (3 rows after vartitle is seen);
178 if find(line, 'class="vartitle"') then do;
179
180 pickup=_n_+3;
181
182 set doc point=pickup;
183 Name=
184 strip(tranwrd(tranwrd(line, '<dd class="info">', ""), "</dd>", ""));
185
186 output;
187 end;
188
189 %* Get Values and Label (1 row after value) ;
190 if find(line, 'scope="row" class="values"') then do;
191
192 pickup1=_n_;
193 pickup2=_n_+1;
194
195 set doc point=pickup1;
196 Value=
197 strip(tranwrd(tranwrd(line, '<td scope="row" class="values">', ""), "</td>", ""));
198
199 set doc point=pickup2;
200 Code=
201 strip(tranwrd(tranwrd(line, '<td class="values">', ""), "</td>", ""));
202 %* Change apostrophe to do not;
203 Code=tranwrd(Code, "Don't", "Do not");
204 output;
205 end;
206
207 run;
208
209 %* If CODEBOOK has no observations put warning to check site;
210 %let dsid=%sysfunc(open(codebook));
211 %let nobs =%sysfunc(attrn(&dsid,nobs));
212 %let rc=%sysfunc(close(&DSID));
213
214 %if &nobs eq 0 %then %put WARNING: No data read from https://wwwn.cdc.gov/Nchs/Nhanes/&cycle/&file..htm. Please check existance of site.;
215
216
217 %* Data clean up ;
218 data file&i (keep=FmtName start end label type);
219 length FmtName start end Label $ 32767 type $ 1;
220 set codebook;
221
222 %* Carry VarName to all rows ;
223 retain VarName;
224 if Name ne "" then VarName=Name;
225
226 %*Missing blanks;
227 if Value = "%str(&)lt; blank %str(&)gt;" then Value = " ";
228
229 %* Do not want SEQN in code book, also delete if value and code are blank;
230 %* If requested, omit formats for missing values;
231 if VarName="SEQN" or (Value="" and Code="")
232 %if %upcase(&omitmissing) in (Y YES 1) %then %do;
233 or (Value in ("." " ") and Code = "Missing")
234 %end;
235 then delete;
236
237 %* Format Name ;
238 FmtName=strip(VarName)||"F";
239
240 %* Create Start and End Values ;
241 if Code="Range of Values" then do;
242 start=strip(scan(Value,1, " "));
243 end=strip(scan(Value,3, " "));
244 label="Valid value";
245 type="N";
246 end;
247 else do;
248 start=strip(Value);
249 end=start;
250 label=Code;
251 if notdigit(compress(Value, ".-")) le length(compress(Value, ".-"))
252 then Type = "C";
253 else Type = "N";
254 end;
255 run;
256
257%end; **end to do i to numfiles;
258
259%* Concatante all files;
260data cntlin;
261 set %do i=1 %to &NumFiles; file&i %end; ;
262run;
263
264%* Read as formats ;
265proc format cntlin=cntlin; run;
266
267%* Create Catalog - this way I can control the name of the catalog;
268proc catalog cat=work.Formats;
269 copy out=&fmtlibrary..&fmtcatalog;
270run;
271
272%* Print formats if requested;
273%if %upcase(&prints) in (1 Y YES) %then %do;
274 proc format library=&fmtlibrary..&fmtcatalog fmtlib; run;
275%end;
276
277%* If requested, create a code based format file;
278%if %length(&fmtcodefile)>0 %then %do;
279
280 %* Add order so that I can print in same order;
281 data cntlin;
282 set cntlin;
283 FmtOrder=_n_;
284 run;
285
286 %* List of variables;
287 proc sql noprint;
288 select FmtName, min(fmtorder) as fmtOrder
289 into :flist separated by ' ' , :dummy
290 from cntlin
291 group by fmtname
292 order by fmtorder;
293 %let NumFmts = &SQLObs.;
294 quit;
295 %put &numfmts;
296
297 %* Create macro vars with info needed ;
298 %* (fmtname, levels, # of levels);
299 %do i=1 %to &NumFmts;
300 **fmt;
301 %let fmt=%scan(&flist, &i);
302
303 **fmt levels and labels;
304 proc sql noprint;
305 select left(trim(fmtname)), start, end, label
306 into :&fmt.1-:&fmt.9999, :&fmt.s1-:&fmt.s9999,
307 :&fmt.e1-:&fmt.e9999, :&fmt.lab1-:&fmt.lab9999
308 from cntlin
309 where fmtname=upcase("&fmt");
310 **fmt num levels;
311 %let count_&fmt = &SqlObs.;
312 quit;
313
314 %end; **end to do i to numfmts;
315
316 **Create File;
317 data _null_;
318 file &fmtcodefile;
319 put "proc format; ";
320 %do i=1 %to &NumFmts;
321 %let fmt=%scan(&flist, &i);
322
323 **formats;
324 put " value &fmt." ;
325 %do j=1 %to &&count_&fmt.;
326 %if "&&&fmt.s&j" eq "&&&fmt.e&j" %then %do;
327 put " &&&fmt.s&j = '&&&fmt.lab&j'";
328 %end;
329 %if "&&&fmt.s&j" ne "&&&fmt.e&j" %then %do;
330 put " &&&fmt.s&j - &&&fmt.e&j = '&&&fmt.lab&j'";
331 %end;
332
333 %end;
334
335 put " ;";
336 %end;
337
338 put "run;";
339 put "quit;";
340 run;
341
342
343%end; **end to if fmtcodefile;
344
345%* Clear work lib;
346proc datasets lib=work memtype=data nolist;
347 delete doc codebook cntlin
348 %do i=1 %to &NumFiles; file&i %end; ;
349run; quit;

351%*****************************************************************************;
352%* CLOSE MACRO ;
353%*****************************************************************************;
354%mend nhanesFmtScrape;
