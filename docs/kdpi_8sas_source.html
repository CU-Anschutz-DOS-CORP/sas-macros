36 @returns
37 The following columns will be added to the data set being created:
38 kdri_rao, kdri_med, and kdpi
39
40 @param donType Variable specifying donor type coded as C for cadaveric
41 or L for living
42
43 @param age Variable containing donor age
44
45 @param hgt_cm Variable containing donor height in centimeters
46
47 @param wgt_kg Variable containing donor weight in kilograms
48
49 @param black Variable indicating whether the donor is of Black race
50 @n This must be coded as 1 = black and 0 = other race
51
52 @param htn Variable indicating whether donor has history of hypertension
53 @n This must be coded as 1 = yes and 0 = no
54
55 @param diabetes Variable indicating whether donor has history of diabetes
56 @n This must be coded as 1 = yes and 0 = no
57
58 @param codCVA Variable indicating whether the donor died due to CVA or no
59 @n This must be coded as 1 = CVA and 0 = other cause of death

61 @param creatinine Variable containing donor serum creatinine in mg/dL
62
63 @param hcv Variable indicating whether the donor is HCV positive
64 @n This must be coded as 1 = yes and 0 = no
65
66 @param dcd Variable indicating the donor DCD status
67 @n This must be coded as 1 = DCD and 0 = DBD
68
69 @scalingFactor A single number representing the scaling factor for converting KDRI RAO
70 to scaled KDRI
71 @n This is obtained from the KDPI mapping table for the reference population
72 of choice.
73
74 @kdpiInformat The name of the informat that contains the KDPI mapping table for the
75 reference population of choice.
76
77 @param prefix Optional prefix for variable names
78 @n For example, if prefix = d_ then the 3 variables created will be d_kdpi_rao d_kdpi_med d_kdpi
79
80 @note
81 @parblock
82 @li This macro will only run from within a data step.
83
84 @li KDPI is only calcualted for cadaveric donors donTyp='C'
85
86 @li The user must have defined an informat with the KDPI mapping table of choice.
87
88 @li KDRIrao (kdri_rao) is interpreted as the relative risk of post-transplant graft failure
89 for this donor compared to a reference donor (age=40 years, non-African American, etc.) as
90 defined in Rao, et al.
91
92 @li KDRImedian (kdri_med) is interpreted as the relative risk of post-transplant graft failure

94 in the reference population year.
95
96 @li KDRI is a measure of relative risk of post-transplant graft failure in a cumulative
97 percentage scale.
98
99 @li Example interpretation: The estimated risk of kidney graft failure from this donor is higher
100 than <KDPI>% of kidney donors recovered in <reference population year> and <KDPImedian> times

102

106 @n Note that the calculator uses the prior year as the reference population (e.g. if you use it in 2023,

161%macro kdpi(
162 donType =
163 ,age =
164 ,hgt_cm =
165 ,wgt_kg =
166 ,black =
167 ,htn =
168 ,diabetes =
169 ,codCVA =
170 ,creatinine =
171 ,hcv =
172 ,dcd =
173 ,scalingFactor =
174 ,kdpiInformat =
175 ,prefix =
176);
177 options fmterr; **This will generate an error if the stated format does not exist;
178 if &donType = 'C' and
179 nmiss(&age, &hgt_cm, &wgt_kg, &black, &htn, &diabetes, &codCVA, &creatinine, &hcv, &dcd) = 0
180 then do;
181 &prefix.kdri_rao =
182 exp(
183 (0.0128 * (&age - 40)) -
184 (0.0194 * (&age - 18) * (&age < 18)) +
185 (0.0107 * (&age - 50) * (&age > 50)) -
186 (0.0464 * divide(&hgt_cm - 170, 10)) -
187 (0.0199 * divide(&wgt_kg - 80, 5) * (&wgt_kg < 80)) +
188 (0.1790 * &black) +
189 (0.1260 * &htn) +
190 (0.1300 * &diabetes) +
191 (0.0881 * &codCVA) +
192 (0.2200 * (&creatinine - 1)) -
193 (0.2090 * (&creatinine - 1.5) * (&creatinine > 1.5)) +
194 (0.2400 * &hcv) +
195 (0.1330 * &dcd)
196 );
197
198 &prefix.kdri_med = divide(&prefix.kdri_rao, &scalingFactor);
199
200 &prefix.kdpi = input(&prefix.kdri_med, &kdpiInformat);
201 end;
202 options nofmterr;
203
204%mend kdpi;
